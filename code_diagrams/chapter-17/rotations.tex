% rotation formulas

$\begin{bmatrix}
    x \\
    y
\end{bmatrix}$

$\begin{bmatrix}
    x * \cos(theta) \\
    x * \sin(theta)
\end{bmatrix}$

$\begin{bmatrix}
    - y * \sin(theta) \\
     y * \cos(theta)
\end{bmatrix}$

$\begin{bmatrix}
    x * \cos(theta) - y * \sin(theta) \\
    x * \sin(theta) + y * \cos(theta)
\end{bmatrix}$
